import os
from dotenv import load_dotenv

# Load environment variables
load_dotenv()

    browser="chrome",
    ai_provider="openai",
    ai_config={
        "api_key": os.getenv("OPENAI_API_KEY"),
        "model": "gpt-4",
        "max_tokens": 2048,
        "temperature": 0.7
    }
)

# Start the agent
await agent.start()

# Use natural language commands

    "Find the search box and search for 'browser automation'"
)
print(result)

async def intelligent_data_extraction():
    agent = BrowserAgent(ai_provider="openai")
    await agent.start()
    
    # Navigate to a product listing page
    await agent.navigate("https://ecommerce-site.com/products")
    
    # Extract structured product data
    products = await agent.ai_extract("""
        Extract all products from this page and return as JSON with:
        {
            "products": [
                {
                    "name": "product name",
                    "price": "price as number",
                    "currency": "currency code",
                    "rating": "rating as number",
                    "reviews_count": "number of reviews",
                    "image_url": "product image URL",
                    "availability": "in stock/out of stock",
                    "features": ["list of key features"]
                }
            ]
        }
    """)
    
    # Process and validate the extracted data
    import json
    product_data = json.loads(products)
    
    # Filter and sort products
    filtered_products = await agent.ai_extract(f"""
        From this product data: {product_data}
        
        Filter and return only products that:
        1. Are in stock
        2. Have rating >= 4.0
        3. Have at least 10 reviews
        
        Sort by rating (highest first), then by price (lowest first)
    """)
    
    return json.loads(filtered_products)

from browser_agent_mcp import MCPServer
from browser_agent import BrowserAgent
import asyncio

async def start_mcp_server():
    # Initialize Browser Agent
    agent = BrowserAgent(
        browser="chrome",
        headless=True,
        ai_provider="openai"
    )
    
    # Create MCP server
    server = MCPServer(
        name="browser-agent-mcp",
        version="1.0.0",
        browser_agent=agent
    )
    
    # Register tools
    server.register_tool("navigate", agent.navigate)
    server.register_tool("extract_data", agent.ai_extract)
    server.register_tool("interact", agent.ai_interact)
    
    # Register resources
    server.register_resource("current_page", agent.get_page_source)
    server.register_resource("page_screenshot", agent.take_screenshot)
    
    # Start server
    await server.start(host="localhost", port=8080)
    print("MCP Server started on http://localhost:8080")

if __name__ == "__main__":
    asyncio.run(start_mcp_server())

from mcp_client import MCPClient
import asyncio

async def use_browser_agent_via_mcp():
    # Connect to Browser Agent MCP server
    client = MCPClient("http://localhost:8080")
    await client.connect()
    
    # Discover available tools
    tools = await client.list_tools()
    print(f"Available tools: {[tool['name'] for tool in tools]}")
    
    # Use the navigate tool
    result = await client.call_tool(
        "navigate",
        {"url": "https://example.com"}
    )
    print(f"Navigation result: {result}")
    
    # Extract data using AI
    data = await client.call_tool(
        "extract_data",
        {"prompt": "Extract the main heading and all links from this page"}
    )
    print(f"Extracted data: {data}")
    
    # Interact with the page
    interaction = await client.call_tool(
        "interact",
        {"instruction": "Click on the 'About' link if it exists"}
    )
    print(f"Interaction result: {interaction}")
    
    # Get page screenshot
    screenshot = await client.get_resource("page_screenshot")
    with open("page_screenshot.png", "wb") as f:
        f.write(screenshot)
    
    await client.disconnect()

if __name__ == "__main__":

üîÑ Multi-Agent Orchestration

Coordinate multiple Browser Agent instances for complex workflows.

Python 
from browser_agent import BrowserAgent
import asyncio

class MultiAgentOrchestrator:
    def __init__(self):
        self.agents = {}
        
    async def create_agent(self, name, config):
        agent = BrowserAgent(**config)
        await agent.start()
        self.agents[name] = agent
        return agent
    
    async def parallel_execution(self, tasks):
        results = await asyncio.gather(*[
            self.execute_task(task) for task in tasks
        ])
        return results
    
    async def execute_task(self, task):
        agent = self.agents[task['agent']]
        return await agent.ai_interact(task['instruction'])

# Usage
orchestrator = MultiAgentOrchestrator()

# Create specialized agents
await orchestrator.create_agent('scraper', {
    'browser': 'chrome',
    'headless': True,
    'ai_provider': 'openai'
})

await orchestrator.create_agent('tester', {
    'browser': 'firefox',
    'headless': False,
    'ai_provider': 'claude'
})

# Execute parallel tasks
tasks = [
    {'agent': 'scraper', 'instruction': 'Scrape product data'},
    {'agent': 'tester', 'instruction': 'Test checkout process'}
]

results = await orchestrator.parallel_execution(tasks)

class AIProcessingPipeline:
    def __init__(self, agent):
        self.agent = agent
        self.models = {
            'vision': 'gemini-pro-vision',
            'reasoning': 'claude-3-sonnet',
            'action': 'gpt-4'
        }
    
    async def process_page(self, url):
        await self.agent.navigate(url)
        
        # Step 1: Visual analysis
        screenshot = await self.agent.take_screenshot()
        visual_analysis = await self.agent.ai_extract(
            "Analyze this screenshot and describe the page layout",
            model=self.models['vision']
        )
        
        # Step 2: Content reasoning
        content_analysis = await self.agent.ai_extract(
            f"Based on this visual analysis: {visual_analysis}, "
            "reason about the best strategy to extract data",
            model=self.models['reasoning']
        )
        
        # Step 3: Execute actions
        result = await self.agent.ai_interact(
            f"Execute this strategy: {content_analysis}",
            model=self.models['action']
        )
        
        return {
            'visual_analysis': visual_analysis,
            'strategy': content_analysis,
            'result': result
        }

            await agent.close()

        return True  # Simplified
    
    async def send_alert(self, target, data):

            'timestamp': datetime.now(),
            'target': target['name'],
            'data': data,
